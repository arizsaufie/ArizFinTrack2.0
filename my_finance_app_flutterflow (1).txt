// lib/main.dart
import 'package:flutter/material.dart';
import 'package:shared_preferences/shared_preferences.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatefulWidget {
  const MyApp({super.key});

  @override
  State<MyApp> createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  bool isEnglish = true;

  void toggleLanguage(bool value) {
    setState(() => isEnglish = value);
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'My Finance RM',
      theme: ThemeData(primarySwatch: Colors.green),
      home: PinPage(isEnglish: isEnglish, onLanguageChange: toggleLanguage),
    );
  }
}

class T {
  static String t(bool en, String eng, String bm) => en ? eng : bm;
}

class PinPage extends StatefulWidget {
  final bool isEnglish;
  final Function(bool) onLanguageChange;

  const PinPage({super.key, required this.isEnglish, required this.onLanguageChange});

  @override
  State<PinPage> createState() => _PinPageState();
}

class _PinPageState extends State<PinPage> {
  final TextEditingController pin = TextEditingController();
  final String savedPin = '1234';

  void unlock() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString('pin', savedPin);
    if (pin.text == savedPin) {
      Navigator.pushReplacement(
        context,
        MaterialPageRoute(builder: (_) => Dashboard(isEnglish: widget.isEnglish)),
      );
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Padding(
        padding: const EdgeInsets.all(32),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Row(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                const Text('EN'),
                Switch(
                  value: !widget.isEnglish,
                  onChanged: (v) => widget.onLanguageChange(!v),
                ),
                const Text('BM'),
              ],
            ),
            const SizedBox(height: 20),
            Text(T.t(widget.isEnglish, 'Enter PIN', 'Masukkan PIN'), style: const TextStyle(fontSize: 20)),
            TextField(controller: pin, obscureText: true, keyboardType: TextInputType.number),
            const SizedBox(height: 20),
            ElevatedButton(onPressed: unlock, child: Text(T.t(widget.isEnglish, 'Unlock', 'Buka')))
          ],
        ),
      ),
    );
  }
}

class Debt {
  String name;
  double original;
  double remaining;
  double installment;
  DateTime dueDate;

  Debt(this.name, this.original, this.installment, this.dueDate) : remaining = original;

  void pay(double amount) {
    remaining -= amount;
    if (remaining < 0) remaining = 0;
  }

  double get progress => (original - remaining) / original;
  String get status => remaining == 0 ? 'Done' : 'Active';
}

class Dashboard extends StatefulWidget {
  final bool isEnglish;
  const Dashboard({super.key, required this.isEnglish});

  @override
  State<Dashboard> createState() => _DashboardState();
}

class _DashboardState extends State<Dashboard> {
  final List<Debt> debts = [];

  void addDebt() {
    setState(() {
      debts.add(Debt(T.t(widget.isEnglish, 'Car Loan', 'Pinjaman Kereta'), 30000, 600, DateTime.now().add(const Duration(days: 30))));
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text(T.t(widget.isEnglish, 'Dashboard', 'Papan Pemuka'))),
      floatingActionButton: FloatingActionButton(onPressed: addDebt, child: const Icon(Icons.add)),
      body: ListView(
        padding: const EdgeInsets.all(16),
        children: debts.map((d) => Card(
          child: Padding(
            padding: const EdgeInsets.all(12),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(d.name, style: const TextStyle(fontWeight: FontWeight.bold)),
                Text(T.t(widget.isEnglish, 'Original: RM ${d.original}', 'Jumlah Asal: RM ${d.original}')),
                Text(T.t(widget.isEnglish, 'Remaining: RM ${d.remaining}', 'Baki: RM ${d.remaining}')),
                Text(T.t(widget.isEnglish, 'Installment: RM ${d.installment}', 'Ansuran: RM ${d.installment}')),
                Text(T.t(widget.isEnglish, 'Due: ${d.dueDate.day}/${d.dueDate.month}/${d.dueDate.year}', 'Due: ${d.dueDate.day}/${d.dueDate.month}/${d.dueDate.year}')),
                LinearProgressIndicator(value: d.progress),
                Text('${(d.progress * 100).toStringAsFixed(1)}% ${T.t(widget.isEnglish, 'completed', 'selesai')}'),
                Text(T.t(widget.isEnglish, 'Status: ${d.status}', 'Status: ${d.status}'))
              ],
            ),
          ),
        )).toList(),
      ),
    );
  }
}
